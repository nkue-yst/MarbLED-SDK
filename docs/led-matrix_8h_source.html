38 // The RGB matrix provides the framebuffer and the facilities to constantly
39 // update the LED matrix.
40 //
41 // This implement the Canvas interface that represents the display with
42 // (led_cols * chained_displays)x(rows * parallel_displays) pixels.
43 //
44 // If can do multi-buffering using the CreateFrameCanvas() and SwapOnVSync()
45 // methods. This is useful for animations and to prevent tearing.
46 //
47 // If you arrange the panels in a different way in the physical space, write
48 // a CanvasTransformer that does coordinate remapping and which should be added
49 // to the transformers, like with UArrangementTransformer in demo-main.cc.

204  // Create a new buffer to be used for multi-buffering. The returned new
205  // Buffer implements a Canvas with the same size of thie RGBMatrix.
206  // You can use it to draw off-screen on it, then swap it with the active
207  // buffer using SwapOnVSync(). That would be classic double-buffering.
208  //
209  // You can also create as many FrameCanvas as you like and for instance use
210  // them to pre-fill scenes of an animation for fast playback later.
211  //
212  // The ownership of the created Canvases remains with the RGBMatrix, so you
213  // don't have to worry about deleting them (but you also don't want to create
214  // more than needed as this will fill up your memory as they are only deleted
215  // when the RGBMatrix is deleted).
216  FrameCanvas *CreateFrameCanvas();
217 
218  // This method waits to the next VSync and swaps the active buffer with the
219  // supplied buffer. The formerly active buffer is returned.
220  //
221  // If you pass in NULL, the active buffer is returned, but it won't be
222  // replaced with NULL. You can use the NULL-behavior to just wait on
223  // VSync or to retrieve the initial buffer when preparing a multi-buffer
224  // animation.
225  //
226  // The optional "framerate_fraction" parameter allows to choose which
227  // multiple of the global frame-count to use. So it slows down your animation
228  // to an exact integer fraction of the refresh rate.
229  // Default is 1, so immediately next available frame.
230  // (Say you have 140Hz refresh rate, then a value of 5 would give you an
231  // 28Hz animation, nicely locked to the refresh-rate).
232  // If you combine this with Options::limit_refresh_rate_hz you can create
233  // time-correct animations.
234  FrameCanvas *SwapOnVSync(FrameCanvas *other, unsigned framerate_fraction = 1);
235

268  //-- GPIO interaction.
269  // This library uses the GPIO pins to drive the matrix; this is a safe way
270  // to request the 'remaining' bits to be used for user purposes.
271 
272  // Request user readable GPIO bits.
273  // This function allows you to request pins you'd like to read with
274  // AwaitInputChange().
275  // Only bits that are not already in use for reading or wrtiting
276  // by the matrix are allowed.
277  // Input is a bitmap of all the GPIO bits you're interested in; returns all
278  // the bits that are actually available.
279  uint64_t RequestInputs(uint64_t all_interested_bits);
280 
281  // This function will return whenever the GPIO input pins
282  // change (pins that are not already in use for output, that is) or the
283  // timeout is reached. You need to have reserved the inputs with
284  // matrix->RequestInputs(...) first (e.g.
285  // matrix->RequestInputs((1<<25)|(1<<24));
286  //
287  // A positive timeout waits the given amount of milliseconds for a change
288  // (e.g. a button-press) to occur; if there is no change, it will just
289  // return the last value.
290  // If you just want to know how the pins are right now, call with zero
291  // timeout.
292  // A negative number waits forever and will only return if there is a change.
293  //
294  // This function only samples between display refreshes so polling some
295  // input does not generate flicker and provide a convenient change interface.
296  //
297  // Returns the bitmap of all GPIO input pins.
298  uint64_t AwaitInputChange(int timeout_ms);
299

301  // This allows to request a bitmap of GPIO-bits to be used by the user for
302  // writing.
303  // Only bits that are not already in use for reading or wrtiting
304  // by the matrix are allowed.
305  // Returns the subset bits that are _actually_ available,
306  uint64_t RequestOutputs(uint64_t output_bits);

393 // Runtime options to simplify doing common things for many programs such as
394 // dropping privileges and becoming a daemon.
395 struct RuntimeOptions {
396  RuntimeOptions();
397 
398  int gpio_slowdown; // 0 = no slowdown. Flag: --led-slowdown-gpio
399 
400  // ----------
401  // If the following options are set to disabled with -1, they are not
402  // even offered via the command line flags.
403  // ----------
404 
405  // Thre are three possible values here
406  // -1 : don't leave choise of becoming daemon to the command line parsing.
407  // If set to -1, the --led-daemon option is not offered.
408  // 0 : do not becoma a daemon, run in forgreound (default value)
409  // 1 : become a daemon, run in background.
410  //
411  // If daemon is disabled (= -1), the user has to call
412  // RGBMatrix::StartRefresh() manually once the matrix is created, to leave
413  // the decision to become a daemon
414  // after the call (which requires that no threads have been started yet).
415  // In the other cases (off or on), the choice is already made, so the thread
416  // is conveniently already started for you.
417  int daemon; // -1 disabled. 0=off, 1=on. Flag: --led-daemon
418 
419  // Drop privileges from 'root' to 'daemon' once the hardware is initialized.
420  // This is usually a good idea unless you need to stay on elevated privs.
421  int drop_privileges; // -1 disabled. 0=off, 1=on. flag: --led-drop-privs
422 
423  // By default, the gpio is initialized for you, but if you run on a platform
424  // not the Raspberry Pi, this will fail. If you don't need to access GPIO
425  // e.g. you want to just create a stream output (see content-streamer.h),
426  // set this to false.
427  bool do_gpio_init;
428 };
