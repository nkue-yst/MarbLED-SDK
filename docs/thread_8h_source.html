23 // Simple thread abstraction.
24 class Thread {
25 public:
26  Thread();
27 
28  // The destructor waits for Run() to return so make sure it does.
29  virtual ~Thread();
30 
31  // Wait for the Run() method to return.
32  void WaitStopped();
33 
34  // Start thread. If realtime_priority is > 0, then this will be a
35  // thread with SCHED_FIFO and the given priority.
36  // If cpu_affinity is set !=, chooses the given bitmask of CPUs
37  // this thread should have an affinity to.
38  // On a Raspberry Pi 1, this doesn't matter, as there is only one core,
39  // Raspberry Pi 2 can has 4 cores, so any combination of (1<<0) .. (1<<3) is
40  // valid.
41  virtual void Start(int realtime_priority = 0, uint32_t cpu_affinity_mask = 0);
42 
43  // Override this to do the work.
44  //
45  // This will be called in a thread once Start() has been called. You typically
46  // will have an endless loop doing stuff.
47  //
48  // It is a good idea to provide a way to communicate to the thread that
49  // it should stop (see ThreadedCanvasManipulator for an example)
50  virtual void Run() = 0;
51 
52 private:
53  static void *PthreadCallRun(void *tobject);
54  bool started_;
55  pthread_t thread_;
56 };
