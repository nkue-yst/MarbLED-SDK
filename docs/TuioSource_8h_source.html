132 void setSourceString(std::string src_string) {
133
134 if (src_string.length()==0) return;
135 char *data = strdup(src_string.c_str());
136
137 char *name_inst = strtok(data, "@");
138
139 char *addr = strtok(NULL, "@");
140 if (addr!=NULL) source_address = std::string(addr);
141 else source_address = (char*)"0x7F000001";
142
143 char *name = strtok(name_inst, ":");
144 source_name = std::string(name);
145
146 char *inst = strtok(NULL, ":");
147 if (inst!=NULL) source_instance = atoi(inst);
148 else source_instance = 0;
149 }
