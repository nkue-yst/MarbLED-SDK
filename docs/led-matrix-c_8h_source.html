48 struct RGBLedMatrixOptions {
49  /*
50  * Name of the hardware mapping used. If passed NULL here, the default
51  * is used.
52  */
53  const char *hardware_mapping;
54 
55  /* The "rows" are the number of rows supported by the display, so 32 or 16.
56  * Default: 32.
57  * Corresponding flag: --led-rows
58  */
59  int rows;
60 
61  /* The "cols" are the number of columns per panel. Typically something
62  * like 32, but also 64 is possible. Sometimes even 40.
63  * cols * chain_length is the total length of the display, so you can
64  * represent a 64 wide display as cols=32, chain=2 or cols=64, chain=1;
65  * same thing.
66  * Flag: --led-cols
67  */
68  int cols;
69 
70  /* The chain_length is the number of displays daisy-chained together
71  * (output of one connected to input of next). Default: 1
72  * Corresponding flag: --led-chain
73  */
74  int chain_length;
75 
76  /* The number of parallel chains connected to the Pi; in old Pis with 26
77  * GPIO pins, that is 1, in newer Pis with 40 interfaces pins, that can
78  * also be 2 or 3. The effective number of pixels in vertical direction is
79  * then thus rows * parallel. Default: 1
80  * Corresponding flag: --led-parallel
81  */
82  int parallel;
83 
84  /* Set PWM bits used for output. Default is 11, but if you only deal with
85  * limited comic-colors, 1 might be sufficient. Lower require less CPU and
86  * increases refresh-rate.
87  * Corresponding flag: --led-pwm-bits
88  */
89  int pwm_bits;
90 
91  /* Change the base time-unit for the on-time in the lowest
92  * significant bit in nanoseconds.
93  * Higher numbers provide better quality (more accurate color, less
94  * ghosting), but have a negative impact on the frame rate.
95  * Corresponding flag: --led-pwm-lsb-nanoseconds
96  */
97  int pwm_lsb_nanoseconds;
98 
99  /* The lower bits can be time-dithered for higher refresh rate.
100  * Corresponding flag: --led-pwm-dither-bits
101  */
102  int pwm_dither_bits;
103 
104  /* The initial brightness of the panel in percent. Valid range is 1..100
105  * Corresponding flag: --led-brightness
106  */
107  int brightness;
108 
109  /* Scan mode: 0=progressive, 1=interlaced
110  * Corresponding flag: --led-scan-mode
111  */
112  int scan_mode;
113 
114  /* Default row address type is 0, corresponding to direct setting of the
115  * row, while row address type 1 is used for panels that only have A/B,
116  * typically some 64x64 panels
117  */
118  int row_address_type; /* Corresponding flag: --led-row-addr-type */
119 
120  /* Type of multiplexing. 0 = direct, 1 = stripe, 2 = checker (typical 1:8)
121  */
122  int multiplexing;
123 
124  /* In case the internal sequence of mapping is not "RGB", this contains the
125  * real mapping. Some panels mix up these colors.
126  */
127  const char *led_rgb_sequence; /* Corresponding flag: --led-rgb-sequence */
128 
129  /* A string describing a sequence of pixel mappers that should be applied
130  * to this matrix. A semicolon-separated list of pixel-mappers with optional
131  * parameter.
132  */
133  const char *pixel_mapper_config; /* Corresponding flag: --led-pixel-mapper */
134 
135  /*
136  * Panel type. Typically just NULL, but certain panels (FM6126) require
137  * an initialization sequence
138  */
139  const char *panel_type; /* Corresponding flag: --led-panel-type */
140 
143  /* Allow to use the hardware subsystem to create pulses. This won't do
144  * anything if output enable is not connected to GPIO 18.
145  * Corresponding flag: --led-hardware-pulse
146  */
147  char disable_hardware_pulsing;
148  char show_refresh_rate; /* Corresponding flag: --led-show-refresh */
149  char inverse_colors; /* Corresponding flag: --led-inverse */
150 
151  /* Limit refresh rate of LED panel. This will help on a loaded system
152  * to keep a constant refresh rate. <= 0 for no limit.
153  */
154  int limit_refresh_rate_hz; /* Corresponding flag: --led-limit-refresh */
155 };

161 struct RGBLedRuntimeOptions {
162  int gpio_slowdown; // 0 = no slowdown. Flag: --led-slowdown-gpio
163 
164  // ----------
165  // If the following options are set to disabled with -1, they are not
166  // even offered via the command line flags.
167  // ----------
168 
169  // Thre are three possible values here
170  // -1 : don't leave choise of becoming daemon to the command line parsing.
171  // If set to -1, the --led-daemon option is not offered.
172  // 0 : do not becoma a daemon, run in forgreound (default value)
173  // 1 : become a daemon, run in background.
174  //
175  // If daemon is disabled (= -1), the user has to call
176  // RGBMatrix::StartRefresh() manually once the matrix is created, to leave
177  // the decision to become a daemon
178  // after the call (which requires that no threads have been started yet).
179  // In the other cases (off or on), the choice is already made, so the thread
180  // is conveniently already started for you.
181  int daemon; // -1 disabled. 0=off, 1=on. Flag: --led-daemon
182 
183  // Drop privileges from 'root' to 'daemon' once the hardware is initialized.
184  // This is usually a good idea unless you need to stay on elevated privs.
185  int drop_privileges; // -1 disabled. 0=off, 1=on. flag: --led-drop-privs
186 
187  // By default, the gpio is initialized for you, but if you run on a platform
188  // not the Raspberry Pi, this will fail. If you don't need to access GPIO
189  // e.g. you want to just create a stream output (see content-streamer.h),
190  // set this to false.
191  bool do_gpio_init;
192 };

338 // Utility function: set an image from the given buffer containting pixels.
339 //
340 // Draw image of size "image_width" and "image_height" from pixel at
341 // canvas-offset "canvas_offset_x", "canvas_offset_y". Image will be shown
342 // cropped on the edges if needed.
343 //
344 // The canvas offset can be negative, i.e. the image start can be shifted
345 // outside the image frame on the left/top edge.
346 //
347 // The buffer needs to be organized as rows with columns of three bytes
348 // organized as rgb or bgr. Thus the size of the buffer needs to be exactly
349 // (3 * image_width * image_height) bytes.
350 //
351 // The "image_buffer" parameters contains the data, "buffer_size_bytes" the
352 // size in bytes.
353 //
354 // If "is_bgr" is 1, the buffer is treated as BGR pixel arrangement instead
355 // of RGB with is_bgr = 0.
356 void set_image(struct LedCanvas *c, int canvas_offset_x, int canvas_offset_y,
357  const uint8_t *image_buffer, size_t buffer_size_bytes,
358  int image_width, int image_height,
359  char is_bgr);
